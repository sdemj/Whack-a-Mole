function selectCategory(category) {
  selectedCategoryName = category;
  showPage('pg-subcategory');
  
  const allBtnArea = document.getElementById("all-btn-area");
  const subGrid = document.getElementById("subcategory-menu-grid");
  
  allBtnArea.innerHTML = "";
  subGrid.innerHTML = "";
  selectedSubcategories.clear();

  const subData = CATEGORY_DATA[category];
  const subKeys = Object.keys(subData);

  const allBtn = document.createElement("button");
  allBtn.className = "menu-box-btn full-width";
  allBtn.id = "btn-all-subcategories";
  allBtn.innerHTML = `대분류 전체`;
  allBtn.onclick = () => toggleAllSubcategories(subKeys);
  allBtnArea.appendChild(allBtn);

  subKeys.forEach(subKey => {

    btn.className = "menu-box-btn";
    btn.id = `sub-btn-${subKey}`;

    subGrid.appendChild(btn);
  });

function toggleSubcategory(subKey) {
  const btn = document.getElementById(`sub-btn-${subKey}`);
  if (selectedSubcategories.has(subKey)) {
    selectedSubcategories.delete(subKey);
    if(btn) btn.classList.remove("active");
  } else {
    selectedSubcategories.add(subKey);
    if(btn) btn.classList.add("active");
  }
  
  const subKeys = Object.keys(CATEGORY_DATA[selectedCategoryName]);
  const allBtn = document.getElementById("btn-all-subcategories");
  if (selectedSubcategories.size === subKeys.length) {
    if(allBtn) allBtn.classList.add("active");
  } else {
    if(allBtn) allBtn.classList.remove("active");
  }
  updateLiveWordCount();
}

function toggleAllSubcategories(subKeys) {
  const allBtn = document.getElementById("btn-all-subcategories");
  const isAllSelected = (selectedSubcategories.size === subKeys.length);

  subKeys.forEach(subKey => {
    const btn = document.getElementById(`sub-btn-${subKey}`);
    if (isAllSelected) {
      selectedSubcategories.delete(subKey);
      if(btn) btn.classList.remove("active");
    } else {
      selectedSubcategories.add(subKey);
      if(btn) btn.classList.add("active");
    }
  });

  if (isAllSelected) {
    if(allBtn) allBtn.classList.remove("active");
  } else {
    if(allBtn) allBtn.classList.add("active");
  }
  updateLiveWordCount();
}

function updateLiveWordCount() {
  let count = 0;
  const subData = CATEGORY_DATA[selectedCategoryName];
  selectedSubcategories.forEach(subKey => {
    if(subData[subKey]) count += subData[subKey].length;
  });
  document.getElementById("live-word-counter").textContent = `선택됨: ${count}단어`;
}

function playWelcomeSound(){
  try {
    const A = window.AudioContext || window.webkitAudioContext; if(!A) return;
    const c = new A(), now = c.currentTime, notes = [110,147,196,294,220,440], durs = [.08,.08,.08,.12,.1,.4];
    let t = 0;
    notes.forEach((f,i)=>{
      const o=c.createOscillator(),g=c.createGain();
      o.type=i===notes.length-1?'sawtooth':'triangle'; o.frequency.setValueAtTime(f,now+t);
      if(i===notes.length-1) o.frequency.exponentialRampToValueAtTime(880,now+t+durs[i]);
      g.gain.setValueAtTime(.18,now+t); g.gain.exponentialRampToValueAtTime(.01,now+t+durs[i]);
      o.connect(g); g.connect(c.destination); o.start(now+t); o.stop(now+t+durs[i]); t+=durs[i]-.02
    });
  } catch(e){}
}

function getColors(){
  let res = [];
  let active = masterColors.slice(0, groupCount).map(c => c.hex);
  const base = Math.floor(cellCount / active.length);
  active.forEach(col => {
    for (let j = 0; j < base; j++) res.push(col);
  });
  let extraColors = [...active];
  for (let i = extraColors.length - 1; i > 0; i--) {
    const j = Math.floor(Math.random() * (i + 1));
    [extraColors[i], extraColors[j]] = [extraColors[j], extraColors[i]];
  }
  const remaining = cellCount - res.length;
  for (let i = 0; i < remaining; i++) {
    res.push(extraColors[i]);
  }
  for (let i = res.length - 1; i > 0; i--) {
    const j = Math.floor(Math.random() * (i + 1));
    [res[i], res[j]] = [res[j], res[i]];
  }
  return res;
}

function getWords(){
  let sel=[], tmp=[...words];
  for(let i=0;i<cellCount;i++){
    if(!tmp.length) tmp=[...words];
    sel.push(tmp.splice(Math.floor(Math.random()*tmp.length),1)[0]);
  }
  return sel;
}

    grid.innerHTML=''; revIdx=-1; nextBtn.classList.remove('visible');
    resModal.classList.remove('active'); repModal.classList.remove('active'); resBox.classList.remove('fail');
    pausedMs=0; pauseAt=null;
    
    if(reset){
        roundNum=1; lastRec=null; history=[]; 
        roundDisp.textContent='ROUND 1 / 5';
        document.querySelector('#recDisp span').textContent = '이전 라운드 기록: 없음';

    startTime=Date.now(); timerOn=true;

    for(let i=0; i<cellCount; i++){

        const ov=document.createElement('div'); ov.className='overlay'; ov.id='ov-'+i;
        cell.appendChild(ov); grid.appendChild(cell);
    }
}

function fmt(s){
  if(s===null) return '없음';
  if(s>=60){ const m=Math.floor(s/60), r=(s%60).toFixed(2); return `${m}분 ${r}초`; }
  return `${s.toFixed(2)}초`;
}

function showResult(dur) {
    resTime.textContent = fmt(dur);
    resBtns.innerHTML = '';
    const isFinalRound = (roundNum === 5);

    if (lastRec === null || dur < lastRec) {
        success = true;
        history.push({ round: roundNum, time: dur });
        lastRec = dur;

        if (isFinalRound) {
            resMsg.textContent = '기록 단축 성공! 결과를 확인해요.';
            resBtns.innerHTML = `<button class="mbtn mbtn-next" onclick="exitGame()">결과 보기</button>`;
        } else {
            resMsg.textContent = roundNum === 1 ? '첫 라운드 완료!' : '기록 단축 성공! 다음 단계로 도전하세요.';
            resBtns.innerHTML = `<button class="mbtn mbtn-exit" onclick="exitGame()">종료하기</button>
                                 <button class="mbtn mbtn-next" onclick="handleNext()">다음 라운드</button>`;
        }
        resMsg.style.color = '#cbd5e1';
        resBox.classList.remove('fail');
        nextBtn.classList.add('visible');
    } else {
        success = false;
        resMsg.textContent = `기록 단축 실패... (${fmt(lastRec)}보다 빨라야 함)`;
        resMsg.style.color = '#ef4444';
        resBox.classList.add('fail');
        resBtns.innerHTML = `<button class="mbtn mbtn-exit" onclick="exitGame()">종료하기</button>
                             <button class="mbtn mbtn-retry" onclick="handleNext()">한 번 더 도전!</button>`;
        nextBtn.classList.remove('visible');
    }
    resModal.classList.add('active');
}

function handleNext(){
    resModal.classList.remove('active');
    if(success) {
        if (roundNum >= 5) {
            exitGame();
            return;
        }
        roundNum++;
        roundDisp.textContent = `ROUND ${roundNum} / 5`; 
        document.querySelector('#recDisp span').textContent = `이전 라운드 기록: ${fmt(lastRec)}`;
        initGrid(); 
    } else {
        initGrid(); 
    }
}

function exitGame(){
  resModal.classList.remove('active');
  repBody.innerHTML='';
  if(!history.length){
      repBody.innerHTML='<tr><td colspan="2" style="color:#64748b;text-align:center;padding:16px">기록이 없습니다.</td></tr>';
  } else {
      history.forEach(r=>{
          const tr=document.createElement('tr');
          tr.innerHTML=`<td>ROUND ${r.round}</td><td style="color:#fbbf24">${fmt(r.time)}</td>`;
          repBody.appendChild(tr);
      });
  }
  repModal.classList.add('active');
}

function openHTP() {
    if (page === 'pg-game' && timerOn) {
        pauseAt = Date.now();
        timerOn = false; 
    }
    document.getElementById('htpModal').classList.add('active');
}

function closeHTP() {
    document.getElementById('htpModal').classList.remove('active');
    if (page === 'pg-game' && pauseAt) {
        pausedMs += Date.now() - pauseAt;
        pauseAt = null;
        timerOn = true;
    }
}

window.addEventListener('keydown', e => {
    if(e.code !== 'Space') return;
    if(document.activeElement && document.activeElement.tagName === 'INPUT') return;
    if(page === 'pg-welcome' || page === 'pg-category' || page === 'pg-subcategory') return;

    if(!htpModal.classList.contains('active')) e.preventDefault();
    else return;

    if(page === 'pg-setup') { 
        goToCategoryAfterSetup(); 
        return;
    }
    
    if(page === 'pg-game') { 
        if(resModal.classList.contains('active') || repModal.classList.contains('active')) return;
        if(revIdx < cellCount - 1) {
            revIdx++; 
            const ov = document.getElementById('ov-' + revIdx);
            if(ov) ov.classList.add('revealed');
        } 
        else if(revIdx === cellCount - 1 && timerOn) {
            playSuccess(); 
            const end = Date.now();
            let off = pausedMs; 
            if(pauseAt) off += end - pauseAt;
            timerOn = false; 
            showResult((end - startTime - off) / 1000);
        }
    }
});

    container.style.transform = `translate(-50%, -50%) scale(${scale})`;
}
